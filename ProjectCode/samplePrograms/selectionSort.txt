void main()
{
	int size;

	print("Enter the size of the array");
	size = scan();

	int arr[size];
	print("Enter the array values");
	for( int i = 0 ; i < size ; i++ )
	{
		arr[i] = scan();
	}

	for( int i = 0 ; i < size ; i++ )
	{
		int min = i;
		for( int j = i+1 ; j < size ; j++ )
		{
			if( arr[min] > arr[j] )
			{
				min = j;
			}
		}
		int temp = arr[min];
		arr[min] = arr[i];
		arr[i] = temp;
	}

	for( int i = 0 ; i < size ; i++ )
	{
		print(arr[i]);
	}
}
