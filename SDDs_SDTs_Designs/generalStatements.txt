STL 	= StatementList;
ST		= Statement;
STS		= Statements;
NULL 	= Empty;
AST		= Arithmetic Statement
IFST	= If Statement
FORST	= For Statement
DST		= Declaration Statement
RST		= Return Statement
IOST	= I/O Statement
CFST	= Control Flow Statemet

STL		-> 	"{" 				{
									//push({) to the scope stack
									//given a scope id.
								}
			ST ";" STS			{
									ST.break = STS.break = STL.break;
									ST.continue = STS.continue = STS.continue;
								} 
			"}"					{
									//release all variables with the scope
								}

STS		->	ST ";" STS			{
										ST.break = STS.break = STS.break;
										ST.continue = STS.continue = STS.continue;
									}
			| NULL					{}

ST		->	AST					{}
			| IFST 					{}
			| FORST					{} 
			| DST 					{}	
			| RST 					{}
			| IOST 					{}
			| CFST					{
										CFST.break = ST.break;
										CFST.continue = ST.continue;
									}

CFST	->	"break" ";"				{
										code += goto(CFST.break);	
									}
			"continue" ";"			{
										code += goto(CFST.continue);
									}
